CREATE VIEW park_base AS
SELECT 
    p.park_ID,
    p.park_nev,
    a.park_regio,
    a.park_varmegye,
    a.alapadat_ev,
    a.park_email,
    m.management_email
FROM park_azonosito p
LEFT JOIN alapadat_latest a ON p.park_ID = a.park_ID
LEFT JOIN management_latest m ON p.park_ID = m.park_ID;


1. vallalkozasok_latest
CREATE VIEW vallalkozasok_latest AS
SELECT v.*, v.vallalkozasok_ev AS alapadat_ev
FROM vallalkozasok v
JOIN (
    SELECT park_ID, vallalkozasok_ev, MAX(datum) AS max_datum
    FROM vallalkozasok
    GROUP BY park_ID, vallalkozasok_ev
) m
ON v.park_ID = m.park_ID
AND v.vallalkozasok_ev = m.vallalkozasok_ev
AND v.datum = m.max_datum;

-- 1. Vállalkozások legfrissebb adatai
CREATE VIEW vallalkozasok_latest AS
SELECT *, vallalkozasok_ev AS alapadat_ev
FROM (
    SELECT *, ROW_NUMBER() OVER (
        PARTITION BY park_ID, vallalkozasok_ev
        ORDER BY datum DESC, rowid DESC
    ) AS rn FROM vallalkozasok
) WHERE rn = 1;

-- 2. Terület legfrissebb adatai
