CREATE VIEW vallalkozasok_latest AS
SELECT*,
    vallalkozasok_ev AS alapadat_ev
FROM (
    SELECT *,
           ROW_NUMBER() OVER (
               PARTITION BY park_ID, vallalkozasok_ev
               ORDER BY datum DESC
           ) AS rn
    FROM vallalkozasok
)
WHERE rn = 1;

CREATE VIEW park_base AS
SELECT 
    p.park_ID,
    p.park_nev,

    a.park_regio,
    a.park_varmegye,
    a.alapadat_ev,
    a.park_email,

    m.management_email

FROM park_azonosito p

LEFT JOIN alapadat_latest a 
    ON p.park_ID = a.park_ID

LEFT JOIN management_latest m
    ON p.park_ID = m.park_ID;
