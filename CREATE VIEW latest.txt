A riportok alapját képező egyesített nézet
CREATE VIEW park_base AS
SELECT 
    p.park_ID,
    p.park_nev,
    a.park_regio,
    a.park_varmegye,
    a.alapadat_ev,
    a.park_email
FROM park_azonosito p
LEFT JOIN alapadat_latest a ON p.park_ID = a.park_ID;


-- 1. Vállalkozások legfrissebb adatai
CREATE VIEW vallalkozasok_latest AS
SELECT *, vallalkozasok_ev AS alapadat_ev
FROM (
    SELECT *, ROW_NUMBER() OVER (
        PARTITION BY park_ID, vallalkozasok_ev
        ORDER BY datum DESC, rowid DESC
    ) AS rn FROM vallalkozasok
) WHERE rn = 1;

-- 2. Terület legfrissebb adatai
CREATE VIEW terulet_latest AS
SELECT *, terulet_ev AS alapadat_ev
FROM (
    SELECT *, ROW_NUMBER() OVER (
        PARTITION BY park_ID, terulet_ev
        ORDER BY datum DESC, rowid DESC
    ) AS rn FROM terulet
) WHERE rn = 1;

-- 3. Infrastruktúra-fejlesztés legfrissebb adatai
CREATE VIEW infrastrukturafejlesztes_latest AS
SELECT *, felhasznalas_ev AS alapadat_ev
FROM (
    SELECT *, ROW_NUMBER() OVER (
        PARTITION BY park_ID, felhasznalas_ev
        ORDER BY datum DESC, rowid DESC
    ) AS rn FROM infrastrukturafejlesztes
) WHERE rn = 1;

-- 4. EU támogatások legfrissebb adatai (tartalom szerint is bontva)
CREATE VIEW EU_tamogatas_latest AS
SELECT *, tamogatas_ev AS alapadat_ev
FROM (
    SELECT *, ROW_NUMBER() OVER (
        PARTITION BY park_ID, tamogatas_ev, tamogatas_tartalom
        ORDER BY datum DESC, rowid DESC
    ) AS rn FROM EU_tamogatas
) WHERE rn = 1;

-- 5. Kapcsolatok legfrissebb adatai
CREATE VIEW kapcsolatok_latest AS

FROM (
    SELECT *, ROW_NUMBER() OVER (
        PARTITION BY park_ID, kapcsolatok_ev
        ORDER BY datum DESC, rowid DESC
    ) AS rn FROM kapcsolatok
) WHERE rn = 1;

-- 6. Alapadatok legfrissebb állapota
CREATE VIEW alapadat_latest AS
SELECT * FROM (
    SELECT *, ROW_NUMBER() OVER (
        PARTITION BY park_ID, alapadat_ev
        ORDER BY datum DESC, rowid DESC
    ) AS rn FROM alapadat
) WHERE rn = 1;

-- 7. Management elérhetőségek (emailenként a legfrissebb)
CREATE VIEW management_latest AS
SELECT * FROM (
    SELECT *, ROW_NUMBER() OVER (
        PARTITION BY park_ID, management_email
        ORDER BY datum DESC, rowid DESC
    ) AS rn FROM management
) WHERE rn = 1;

-- 8. Infrastruktúra típusok legfrissebb állapota
CREATE VIEW infrastruktura_latest AS
SELECT * FROM (
    SELECT *, ROW_NUMBER() OVER (
        PARTITION BY park_ID, infra_ID
        ORDER BY datum DESC, rowid DESC
    ) AS rn FROM infrastruktura
) WHERE rn = 1;
