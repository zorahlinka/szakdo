ha van én mező
CREATE INDEX idx_<tabla>_latest
ON <tabla>(park_ID, <ev_oszlop>, datum DESC);

ha több dimenzió van
CREATE INDEX idx_<tabla>_latest
ON <tabla>(park_ID, <dim1>, <dim2>, datum DESC);

ha nincs én (entity)
CREATE INDEX idx_<tabla>_latest
ON <tabla>(park_ID, <entity_ID>, datum DESC);
